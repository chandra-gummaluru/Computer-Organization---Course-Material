\begin{frame}{RAM and ROM}
  \begin{center}
    \Large \highlight{\textbf{RAM and ROM}}
  \end{center}
\end{frame}

\begin{frame}
  Memory comes in two broad kinds. They differ in whether they can be written and whether they keep their contents without power.\\[5mm]
  \begin{center}
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{r@{\hspace{8mm}}c@{\hspace{10mm}}c}
                                  & \highlight{\textbf{RAM}} & \highlight{\textbf{ROM}} \\ \hline
      \graytext{can be written}   & \cmark & \xmark \\
      \graytext{kept without power} & \xmark & \cmark \\
      \graytext{typical use}      & \graytext{working data} & \graytext{fixed programs} \\
    \end{tabular}
  \end{center}
\end{frame}

\begin{frame}
  \highlight{\textbf{RAM}} is used for data that changes while a program runs. Its contents are lost when power is removed.\\[3mm]
  \highlight{\textbf{ROM}} is used for data that must survive a power cycle, such as the instructions that start the machine.
\end{frame}

\begin{frame}
  \begin{center}
    Memory stores both the data a program works on and the program itself. Next we build the processor that reads and runs it.
  \end{center}
\end{frame}
