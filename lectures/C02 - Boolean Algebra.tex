\begin{frame}{Notation}
  \begin{center}
    \Large \highlight{\textbf{notation}}
  \end{center}
\end{frame}

\begin{frame}
  We write the three basic gates as operations on variables:\\[3mm]
  \begin{itemize}
    \item \highlight{a \cdot b} \\
          \graytext{the \texttt{AND} of \ttab{a} and \ttab{b}} \\[2mm]
    \item \highlight{a + b} \\
          \graytext{the \texttt{OR} of \ttab{a} and \ttab{b}} \\[2mm]
    \item \highlight{\overline{a}} \\
          \graytext{the \texttt{NOT} of \ttab{a}, also written $a'$ or $\lnot a$}
  \end{itemize}
  \vspace{2mm}
  A \highlight{\textbf{Boolean expression}} combines these to describe a circuit.
\end{frame}

% ------------------------------------------------------------
% laws
% ------------------------------------------------------------
\begin{frame}{Laws}
  \begin{center}
    \Large \highlight{\textbf{laws}}
  \end{center}
\end{frame}

\begin{frame}
  Each law comes in two forms, one for \texttt{AND} and one for \texttt{OR}.\\[4mm]
  \begin{center}
    \renewcommand{\arraystretch}{1.6}
    \begin{tabular}{r@{\hspace{6mm}}c@{\hspace{10mm}}c}
      \graytext{identity}   & $a \cdot 1 = a$          & $a + 0 = a$          \\
      \graytext{null}       & $a \cdot 0 = 0$          & $a + 1 = 1$          \\
      \graytext{idempotent} & $a \cdot a = a$          & $a + a = a$          \\
      \graytext{complement} & $a \cdot \overline{a} = 0$ & $a + \overline{a} = 1$ \\
    \end{tabular}
  \end{center}
\end{frame}

\begin{frame}
  Expressions can also be reordered and regrouped.\\[4mm]
  \begin{center}
    \renewcommand{\arraystretch}{1.8}
    \begin{tabular}{r@{\hspace{6mm}}l}
      \graytext{commutative} & $a \cdot b = b \cdot a$ \\
      \graytext{associative} & $a \cdot (b \cdot c) = (a \cdot b) \cdot c$ \\
      \graytext{distributive} & $a \cdot (b + c) = a \cdot b + a \cdot c$ \\
      \graytext{involution}   & $\overline{\overline{a}} = a$ \\
    \end{tabular}
  \end{center}
\end{frame}

\begin{frame}
  Two more laws let us delete terms that add nothing. They are worth memorising, since they remove gates.\\[4mm]
  \begin{center}
    \renewcommand{\arraystretch}{2.0}
    \begin{tabular}{r@{\hspace{6mm}}l}
      \graytext{absorption} & $a + a \cdot b = a$ \\
      \graytext{consensus}  & $a \cdot b + \overline{a} \cdot c + b \cdot c = a \cdot b + \overline{a} \cdot c$ \\
    \end{tabular}
  \end{center}
\end{frame}

\begin{frame}
  De Morgan's laws allow us to flip between \texttt{AND}s and \texttt{OR}s.\\[4mm]
  \begin{center}
    \renewcommand{\arraystretch}{2.0}
    \begin{tabular}{r@{\hspace{6mm}}l}
      \graytext{first version} & $\overline{a \cdot b} = \overline{a} + \overline{b}$  \\
      \graytext{second version}  & $\overline{a + b} = \overline{a} \cdot \overline{b}$ \\
    \end{tabular}
  \end{center}
\end{frame}


\begin{frame}
We reduce an expression by applying the laws in turn.\\[3mm]

      \node[or gate US, draw, thick, scale=1.1, transform shape] (orgate) at (4.2,0) {};
      \node[not gate US, draw, thick, scale=1.1, transform shape] (not1) at (-3,-2.8) {};

      \draw[thick] (-5.5,0) -- ++(-0.5,0) node[left, smallgraytext] {\ttab{b}};
      \fill (-5.5,0) circle (1.4pt);
      \draw[thick] (-5.5,0) |- (and1.input 1);
      \draw[thick] (-5.5,0) |- (not1.input);

      \draw[thick] (not1.output) -| (and2.input 1);

      \draw[thick] (-6.7,-1.0) -- ++(-0.5,0) node[left, smallgraytext] {\ttab{a}};
      \fill (-6.7,-1.0) circle (1.4pt);
      \draw[thick] (-6.7,-1.0) |- (and1.input 2);
      \draw[thick] (-6.7,-1.0) |- (and2.input 2);

      $y$ & $= a \cdot b + a \cdot \overline{b}$ & \\
          & $= a \cdot (b + \overline{b})$ & \graytext{distributive} \\
