\end{frame}

\begin{frame}{Combinational Logic}
  \begin{center}
    \Large \highlight{\textbf{combinational logic}}
  \end{center}
\end{frame}

\begin{frame}
  We begin with circuits whose output depends only on their current input.\\[2mm]
  \begin{itemize}
    \item \highlight{\textbf{logic gates}}: \\
          \graytext{the basic operations on binary values} \\[2mm]
    \item \highlight{\textbf{circuit design}}: \\
          \graytext{combining gates to compute any function} \\[2mm]
    \item \highlight{\textbf{arithmetic}}: \\
          \graytext{representing numbers and adding them in hardware}
  \end{itemize}
\end{frame}

\begin{frame}{Sequential Logic}
  \begin{center}
    \Large \highlight{\textbf{sequential logic}}
  \end{center}
\end{frame}

\begin{frame}
  Next we add memory, so that output can depend on the past as well as the present:\\[2mm]
  \begin{itemize}
    \item \highlight{\textbf{flip-flops}}: \\
          \graytext{circuits that store a single bit} \\[2mm]
    \item \highlight{\textbf{registers and counters}}: \\
          \graytext{groups of bits that hold and update values} \\[2mm]
    \item \highlight{\textbf{state machines}}: \\
          \graytext{circuits that move through a sequence of states}
  \end{itemize}
\end{frame}

\begin{frame}{Architecture}
  \begin{center}
    \Large \highlight{\textbf{architecture}}
  \end{center}
\end{frame}

\begin{frame}
  Finally we assemble these parts into a working computer:\\[2mm]
  \begin{itemize}
    \item \highlight{\textbf{memory}}: \\
          \graytext{storing many values and reading them by address} \\[2mm]
    \item \highlight{\textbf{the processor}}: \\
          \graytext{a datapath and control that carry out instructions} \\[2mm]
    \item \highlight{\textbf{assembly language}}: \\
          \graytext{the instructions a program gives the processor}
  \end{itemize}
\end{frame}

\begin{frame}
  \begin{center}
    We will build the whole stack, one layer at a time, starting from a single switch.
  \end{center}
\end{frame}
