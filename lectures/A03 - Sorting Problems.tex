    \graytext{(sorting problems)} \\[-4mm]
    \begin{enumerate}
      \item \highlight{\textbf{build the heap:}} \\
            \graytext{insert each element from the original array into an initially empty heap using \texttt{Queue.push}} \\

      \item \highlight{\textbf{extract elements in order:}} \\
            \graytext{Repeatedly remove the top element using \texttt{Queue.pop} and store the element into the next empty spot of a new array}
    \end{enumerate}
  \end{proofbox}
\end{frame}

\begin{frame}[fragile]
  \begin{proofbox}
    \begin{lstlisting}[style=mypython]
def heap_sort(arr: Array) -> Array:
    # build the queue
    queue = Queue()
    for arr_node in arr:
        queue.push(arr.node.obj, arr_node.obj.key)

    # extract nodes
    sorted_arr = Array()
    for idx in range(queue.size):
        next = queue.pop()
        sorted_arr.insert(next.obj, idx)
    return sorted_arr
\end{lstlisting}
  \end{proofbox}
