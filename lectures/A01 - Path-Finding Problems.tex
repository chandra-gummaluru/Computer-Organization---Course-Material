\begin{frame}
  \begin{thmbox}
    \highlight{\textbf{path-finding problems}}: \\
    \graytext{(discovery, routing, loop detection)} \\[2mm]
    \highlightc{myyellow!60}{\textbf{assumptions}}:
    \begin{itemize}
      \item a weighted graph, $\mathcal{G} = (V,E, \textsf{wgt},\dots)$, is given
      \item $\textsf{wgt}(e) > 0$ for all $e \in E$
      \item a starting vertex, $v^{(0)} \in V$, is specified
      \item a subset $G \subseteq V$ is given denoting goal vertices
    \end{itemize}

    \highlightc{myyellow!60}{\textbf{goals}}:
    \begin{itemize}
      \item find a path, $p$, with $\textsf{org}(p) = v^{(0)}$ and $\textsf{dst}(p) \in G$
      \item if multiple such paths exist, return the one with the least weight
      \item if no such path exists, return \texttt{None}
    \end{itemize}

  \end{thmbox}
\end{frame}

\begin{frame}
  \begin{proofbox}
    \highlight{\textbf{path-finding algorithms}}:
    \begin{enumerate}
      \item \highlightc{myyellow!60}{\textbf{explore an open path}} \\
            \graytext{choose one of the open paths to explore; if there are none, stop and return \texttt{None} (no solution)}

      \item \highlightc{myyellow!60}{\textbf{check for goal:}} \\
            \graytext{if the chosen path is goal-terminating, stop and return the chosen path (solution found)}

      \item \highlightc{myyellow!60}{\textbf{expand/export chosen path:}} \\
            \graytext{expand the chosen path and export them as new open paths}
    \end{enumerate}

  \end{proofbox}
\end{frame}

class Path:
# attributes
    # a list of edges representing the path
    edges: List[Edge]
    
    # the number of edges in the path
    length: Integer
    
    # the cumulative weight of the edges in the path
    weight: Integer

    # the origin of this path
    org: Vertex

    # the destination of this path
    dst: Vertex

\begin{frame}[fragile]
  \begin{thmboxbot}
    \begin{lstlisting}[style=mypython, firstnumber=last]
    def expand(self: Path, e: Edge) -> Path:
        new_path = Path(
            self.edges,
            self.length,
            self.weight,
            self.org,
            self.dst)
            
        if edge.from.key is self.dst.key:
            new_path.edges.append(e)
            new_path.length += 1
            new_path.wgt += edge.wgt
            new_path.dst = edge.to
        return new_path
\end{lstlisting}
  \end{thmboxbot}
\end{frame}

\begin{frame}[fragile]
  \vspace{-5mm}
  \begin{thmboxtop}
    \begin{lstlisting}[style=mypython]
def find_path(graph: Graph, goals: Dict[Vertex*, Vertex], start: Vertex*) -> Optional[Path]:
    # helper
    def _find_path(graph: Graph goals: Dict[Vertex*, Vertex], open: Queue[Path]) -> Optional[Path]:
        if open.size is 0:
            return None
        # explore
        path = open.pop()
        if goals.get(path.dst.key) is not None:
            return path
        for edge in graph.get_edges_from(path.dst.key):
            # expand
            new_path = path.expand(edge)
            # export
            open.push(self._pri(new_path), new_path)
        return self.find_path(self, graph, goals, open)
\end{lstlisting}
  \end{thmboxtop}
\end{frame}

\begin{frame}
  \begin{thmbox}
    Let $\textsf{deg}(v)$ denote the degree of vertex $v$ (the number of edges originating at $v$). Then for any directed graph,
    \[\sum_{v \in V}\textsf{deg}(v) = |E|,\]
  \end{thmbox}
  \begin{proofbox}
    Each edge, $e \in E$, has an origin, $\textsf{org}(e)$ and destination, $\textsf{dst}(e)$, and thus increases the degree of each vertex by one (for a total of two).
  \end{proofbox}
\end{frame}
