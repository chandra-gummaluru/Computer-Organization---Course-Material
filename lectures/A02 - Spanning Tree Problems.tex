\begin{frame}
  \begin{thmbox}
    \highlight{\textbf{spanning tree problems}} \\
    \graytext{(broadcasting)} \\[2mm]
    \highlightc{myyellow!60}{\textbf{assumptions}}:
    \begin{itemize}
      \item a connected, weighted graph, $\mathcal{G} = (V,E, \textsf{wgt},\dots)$, is given
      \item $\textsf{wgt}(e) > 0$ for all $e \in E$ $\textsf{wgt}(e)$
    \end{itemize}
    \\[4mm]
    \highlightc{myyellow!60}{\textbf{goals}}:
    \begin{itemize}
      \item find a connected tree, $\mathcal{T} = (V,T)$ such that $T \subseteq E$
      \item if multiple such trees exist, return the one with the least weight
      \item if no such tree exists, return \texttt{None}
    \end{itemize}
  \end{thmbox}
\end{frame}

    \highlight{\textbf{prim's algorithm}}: \\
    \graytext{(spanning tree problems)} \\[-4mm]
    \begin{enumerate}
      \item \highlightc{myyellow!60}{\textbf{find cheapest child:}} \\
            \color{gray}\footnotesize find the cheapest edge across all vertices on the tree that does not start/end at a vertex already on the tree\color{black}\normalsize
      \item \highlightc{myyellow!60}{\textbf{add edge/vertex:}} \\
            \color{gray}\footnotesize add that edge and the corresponding vertex to the tree\color{black}\normalsize
    \end{enumerate}
  \end{thmbox}

def find_mst_prim(graph: Graph, root: Vertex) -> Tree:
    open = Queue[Edge]()
    for edge in graph.get_vertex(root).out_edges:
        open.push(edge, edge.wgt)
    closed = Dict[Key]()
    mst = Tree()
    mst.add_vertex(root)
    return _find_mst_prim(graph, open, closed, mst)
\end{lstlisting}

\begin{frame}[fragile]
  \begin{thmbox}
    \begin{lstlisting}[style=mypython]
def _find_mst_prim(graph: Graph, open: Queue[Edge], closed: Dict[Boolean], mst: Tree) -> Tree:
    if open.size is 0:
        return mst
    edge = open.pop()
    from = edge.from
    to = edge.to
    if closed.get(to.key) is not True:
        mst.append_vertex(to.key, to.obj, from.key)
        closed.append(edge.to.key, True)
        for new_edge in graph.get_edges_from(edge.to.key):
            if closed_Dict[new_edge.to.key].obj is not True:
                open.push(new_edge, new_edge.wgt)
    return _find_mst_prim(graph, open, closed, mst)
\end{lstlisting}
  \end{thmbox}
\end{frame}

\begin{frame}
  \vspace{-6mm}
  \begin{proofbox}
    \highlight{\textbf{kruskal's algorithm}}: \\
    \graytext{(spanning tree problems)} \\[-4mm]
    \begin{enumerate}
      \item \highlight{\textbf{sort edges:}} \\
            \color{gray}\footnotesize sort the \highlight{\textbf{undirected}} edges in increasing order of weight\color{black}\normalsize
            \begin{table}[h!]
              \centering
              \setlength{\tabcolsep}{4mm} % horizontal padding
              \renewcommand{\arraystretch}{1.2} % vertical spacing
              \begin{tabular}{m{0.6cm}|m{0.6cm}|m{0.6cm}}
                \hline
                \textbf{org} & \textbf{dst} & \textbf{wgt} \\\hline
                \devicenodelabelnomargin{\texttt{B}} &
                \devicenodelabelnomargin{\texttt{D}} &
                1 \\\hline
              \end{tabular}
            \end{table}
            \vspace{-2mm}
            \begin{center}
              $\vdots$
            \end{center}
            \vspace{-1mm}
            \begin{table}[h!]
              \centering
              \setlength{\tabcolsep}{4mm}
              \renewcommand{\arraystretch}{1.2}
              \begin{tabular}{m{0.6cm}|m{0.6cm}|m{0.6cm}}
                \hline
                \devicenodelabelnomargin{\texttt{E}}                                                        &
                \scalebox{0.85}{\raisebox{-0.3\height}{\includegraphics[scale=0.045]{images/computer.png}}} &
                14 \\\hline
              \end{tabular}
            \end{table}
      \item \highlight{\textbf{find edge:}} \\
            \color{gray}\footnotesize iterate through the edges to find the next cheapest one that does not cause a cycle\color{black}\normalsize
      \item \highlight{\textbf{union vertices:}} \\
            \color{gray}\footnotesize add that edge to the spanning tree\color{black}\normalsize
    \end{enumerate}
  \end{proofbox}
\end{frame}

    \begin{lstlisting}[style=mypython]
def find_mst_kruskal(graph: Graph) -> Tree:
    mst = Tree(graph.get_vertices())
    edges = sorted(graph.get_edges(), key=lambda e: e[2])
    uf = UnionFind(graph.get_vertices())
    for (v1, v2, wgt) in edges:
        if uf.find_set(v1) != uf.find_set(v2):
            mst.add_edge(v1, v2, wgt)
            uf.union_sets(v1, v2)
    return mst
\end{lstlisting}
  \end{thmbox}
