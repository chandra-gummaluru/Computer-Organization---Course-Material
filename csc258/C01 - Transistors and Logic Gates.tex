% binary values

\begin{frame}{Binary Values}
  \begin{center}
    \Large \highlight{\textbf{binary values}}
  \end{center}
\end{frame}

\begin{frame}
  A wire in a circuit carries one of two voltage levels. We treat these levels as the values \ttab{0} and \ttab{1}.
  \begin{figure}
    \centering
    \begin{tikzpicture}[>=stealth]
      \draw[very thick] (0,0) -- (2.5,0);
      \node[smallgraytext, below=1mm] at (1.25,0) {low voltage};
      \node[above=1mm] at (1.25,0) {\ttab{0}};

      \draw[very thick] (5,0.6) -- (7.5,0.6);
      \node[smallgraytext, below=1mm] at (6.25,0.6) {high voltage};
      \node[above=1mm] at (6.25,0.6) {\ttab{1}};
    \end{tikzpicture}
  \end{figure}
  Every value a computer stores or computes is built from wires like these.
\end{frame}

% ------------------------------------------------------------
% transistor as a switch
% ------------------------------------------------------------
\begin{frame}{The Transistor}
  \begin{center}
    \Large \highlight{\textbf{the transistor}}
  \end{center}
\end{frame}

\begin{frame}
  A transistor has three terminals. The value on the \highlight{\textbf{control}} terminal decides whether current may flow between the other two.
  \begin{figure}
    \centering
    \begin{tikzpicture}[>=stealth]
      % terminals
      \draw[very thick] (0,0) -- (1.2,0);
      \draw[very thick] (2.8,0) -- (4,0);
      \node[circle, draw, thick, fill=white, minimum size=6mm, inner sep=0pt] (g) at (2,1.3) {};
      \draw[very thick] (2,0.4) -- (g);

      % switch arm (open)
      \draw[very thick] (1.2,0) -- (2.7,0.7);
      \fill (1.2,0) circle (1.5pt);
      \fill (2.8,0) circle (1.5pt);

      \node[smallgraytext, left=1mm] at (0,0) {in};
      \node[smallgraytext, right=1mm] at (4,0) {out};
      \node[smallgraytext, above=1mm] at (g.north) {control};
    \end{tikzpicture}
  \end{figure}
  \begin{center}
    \graytext{when control is \ttab{1} the switch closes and current flows}
  \end{center}
\end{frame}

% ------------------------------------------------------------
% logic gates
% ------------------------------------------------------------
\begin{frame}{Logic Gates}
  \begin{center}
    \Large \highlight{\textbf{logic gates}}
  \end{center}
\end{frame}

\begin{frame}
  Wiring a few transistors together gives a \highlight{\textbf{logic gate}}. A gate takes binary inputs and produces a binary output.\\[2mm]
  We describe a gate by its \highlight{\textbf{truth table}}, which lists the output for every possible input.
\end{frame}

% NOT
\begin{frame}{NOT}
  \begin{minipage}{0.5\textwidth}
    \centering
    \highlight{\textbf{NOT}} \\[1mm]
    \graytext{output is the opposite \\ of the input} \\[6mm]
    \begin{tikzpicture}
      \node[not gate US, draw, thick, scale=1.6] (n) at (0,0) {};
      \draw[thick] (n.input) -- ++(-0.6,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (n.output) -- ++(0.6,0) node[right, smallgraytext] {\ttab{y}};
    \end{tikzpicture}
  \end{minipage}%
  \begin{minipage}{0.5\textwidth}
    \centering
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{c|c}
      \ttab{a} & \ttab{y} \\ \hline
      \ttab{0} & \ttab{1} \\
      \ttab{1} & \ttab{0} \\
    \end{tabular}
  \end{minipage}
\end{frame}

% AND
\begin{frame}{AND}
  \begin{minipage}{0.5\textwidth}
    \centering
    \highlight{\textbf{AND}} \\[1mm]
    \graytext{output is \ttab{1} only when \\ both inputs are \ttab{1}} \\[6mm]
    \begin{tikzpicture}
      \node[and gate US, draw, thick, scale=1.6] (g) at (0,0) {};
      \draw[thick] (g.input 1) -- ++(-0.6,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (g.input 2) -- ++(-0.6,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (g.output) -- ++(0.6,0) node[right, smallgraytext] {\ttab{y}};
    \end{tikzpicture}
  \end{minipage}%
  \begin{minipage}{0.5\textwidth}
    \centering
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{cc|c}
      \ttab{a} & \ttab{b} & \ttab{y} \\ \hline
      \ttab{0} & \ttab{0} & \ttab{0} \\
      \ttab{0} & \ttab{1} & \ttab{0} \\
      \ttab{1} & \ttab{0} & \ttab{0} \\
      \ttab{1} & \ttab{1} & \ttab{1} \\
    \end{tabular}
  \end{minipage}
\end{frame}

% NAND / NOR
\begin{frame}{NAND and NOR}

  \begin{minipage}{0.5\textwidth}

    \begin{tikzpicture}
      \node[nand gate US, draw, thick, scale=1.5] (g) at (0,0) {};

      \draw[thick] (g.input 2) -- ++(-0.5,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (g.output) -- ++(0.5,0) node[right, smallgraytext] {\ttab{y}};

    \graytext{\ttab{1} unless both inputs are \ttab{1}}

  \begin{minipage}{0.5\textwidth}
