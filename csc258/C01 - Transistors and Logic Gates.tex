\begin{frame}
  A wire carries one of two voltage levels. We read a high voltage as \ttab{1} and a low voltage as \ttab{0}.
  \begin{figure}
    \centering
    \begin{tikzpicture}[>=stealth]
      \draw[very thick] (0,0) -- (2.5,0);
      \node[smallgraytext, below=1mm] at (1.25,0) {low voltage};

      \draw[very thick] (5,0.6) -- (7.5,0.6);

      \node[above=1mm] at (6.25,0.6) {\ttab{1}};

  \end{figure}

\begin{frame}{The Transistor}
  \begin{center}
    \Large \highlight{\textbf{the transistor}}
  \end{center}
\end{frame}

\begin{frame}
  A transistor has three terminals. The voltage on the \highlight{\textbf{gate}} decides whether current may flow between the \highlight{\textbf{source}} and the \highlight{\textbf{drain}}.
  \begin{figure}
    \centering
    \begin{tikzpicture}
      \pic at (0,0) {nmos};
      \node[smallgraytext, above=0mm] at (0,0.6) {source};
      \node[smallgraytext, below=0mm] at (0,-0.6) {drain};
      \node[smallgraytext, left=0mm] at (-0.6,0) {gate};
    \end{tikzpicture}
  \end{figure}
  \begin{center}
    \graytext{the gate acts like the handle of a switch}
  \end{center}
\end{frame}

% ------------------------------------------------------------
% driving an output
% ------------------------------------------------------------
\begin{frame}
  To build a gate, every input must connect the output to either the high supply or the ground.\\[3mm]
  \begin{itemize}
    \item a \highlight{\textbf{pull-up}} network connects the output to high \\[1mm]
    \item a \highlight{\textbf{pull-down}} network connects the output to low
  \end{itemize}
  \vspace{3mm}
  \begin{center}
    \graytext{leaving the output unconnected is not the same as driving it low}
  \end{center}
\end{frame}

% ------------------------------------------------------------
% NOT gate
% ------------------------------------------------------------
\begin{frame}{Building a NOT Gate}
  \begin{center}
    \Large \highlight{\textbf{building a NOT gate}}
  \end{center}
\end{frame}

\begin{frame}
  One pMOS pulls the output up and one nMOS pulls it down. When the input is \ttab{0} the top transistor conducts, so the output is \ttab{1}, and the reverse.\\[1mm]
  \begin{minipage}{0.55\textwidth}
    \centering
    \begin{tikzpicture}
      % supply and ground rails
      \node[smallgraytext] at (0,2.15) {high};
      \draw[thick] (-0.4,1.9) -- (0.4,1.9);
      \draw[thick] (0,1.9) -- (0,1.6);
      % pMOS (pull-up)
      \pic at (0,1.0) {pmos};
      % nMOS (pull-down)
      \pic at (0,-1.0) {nmos};
      % connect pMOS drain to nMOS source through the output node
      \draw[thick] (0,0.4) -- (0,-0.4);
      \fill (0,0) circle (1.4pt);
      \draw[thick] (0,0) -- (1.1,0) node[right, smallgraytext] {\ttab{y}};
      % ground
      \draw[thick] (0,-1.6) -- (0,-1.9);
      \draw[thick] (-0.4,-1.9) -- (0.4,-1.9);
      \node[smallgraytext] at (0,-2.15) {low};
      % shared input to both gates
      \draw[thick] (-0.6,1.0) -| (-1.1,0) |- (-0.6,-1.0);
      \draw[thick] (-1.1,0) -- ++(-0.5,0) node[left, smallgraytext] {\ttab{a}};
      \fill (-1.1,0) circle (1.4pt);
    \end{tikzpicture}
  \end{minipage}%
  \begin{minipage}{0.45\textwidth}
    \centering
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{c|c}
      \ttab{a} & \ttab{y} \\ \hline
      \ttab{0} & \ttab{1} \\
      \ttab{1} & \ttab{0} \\
    \end{tabular} \\[6mm]
    \begin{tikzpicture}
      \node[not gate US, draw, thick, scale=1.3] (n) at (0,0) {};
      \draw[thick] (n.input) -- ++(-0.5,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (n.output) -- ++(0.5,0) node[right, smallgraytext] {\ttab{y}};
    \end{tikzpicture} \\[1mm]
    \graytext{its gate symbol}
  \end{minipage}
\end{frame}

% ------------------------------------------------------------
% AND gate

\begin{frame}{Building an AND Gate}
  \begin{center}
    \Large \highlight{\textbf{building an AND gate}}
  \end{center}
\end{frame}

\begin{frame}
  Two nMOS in series pull the output up only when both inputs are \ttab{1}. Two pMOS in parallel pull it down when either input is \ttab{0}.\\[1mm]
  \begin{minipage}{0.55\textwidth}
    \centering
    \begin{tikzpicture}
      % high rail
      \node[smallgraytext] at (0,2.75) {high};
      \draw[thick] (-0.4,2.5) -- (0.4,2.5);
      \draw[thick] (0,2.5) -- (0,2.2);
      % series nMOS (pull-up): A then B
      \pic at (0,1.6) {nmos};
      \draw[thick] (-0.6,1.6) -- ++(-0.35,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (0,1.0) -- (0,0.6);
      \pic at (0,0.0) {nmos};
      \draw[thick] (-0.6,0.0) -- ++(-0.35,0) node[left, smallgraytext] {\ttab{b}};
      % output node
      \draw[thick] (0,-0.6) -- (0,-1.0);
      \fill (0,-1.0) circle (1.4pt);
      \draw[thick] (0,-1.0) -- (2.2,-1.0) node[right, smallgraytext] {\ttab{y}};
      % parallel pMOS (pull-down) to ground
      \pic at (0.9,-2.0) {pmos};
      \pic at (2.0,-2.0) {pmos};
      \draw[thick] (0.9,-1.4) -- (0.9,-1.0);
      \draw[thick] (2.0,-1.4) -- (2.0,-1.0);
      \draw[thick] (0.9,-2.6) -- (0.9,-2.9);
      \draw[thick] (2.0,-2.6) -- (2.0,-2.9);
      \draw[thick] (0.5,-2.9) -- (2.4,-2.9);
      \node[smallgraytext] at (1.45,-3.15) {low};
      \draw[thick] (0.3,-2.0) -- ++(-0.3,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (1.4,-2.0) -- ++(-0.3,0) node[left, smallgraytext] {\ttab{b}};
    \end{tikzpicture}
  \end{minipage}%
  \begin{minipage}{0.45\textwidth}
    \centering
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{cc|c}
      \ttab{a} & \ttab{b} & \ttab{y} \\ \hline
      \ttab{0} & \ttab{0} & \ttab{0} \\
      \ttab{0} & \ttab{1} & \ttab{0} \\
      \ttab{1} & \ttab{0} & \ttab{0} \\
      \ttab{1} & \ttab{1} & \ttab{1} \\
    \end{tabular} \\[6mm]
    \begin{tikzpicture}
      \node[and gate US, draw, thick, scale=1.2] (g) at (0,0) {};
      \draw[thick] (g.input 1) -- ++(-0.4,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (g.input 2) -- ++(-0.4,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (g.output) -- ++(0.4,0) node[right, smallgraytext] {\ttab{y}};
    \end{tikzpicture} \\[1mm]
    \graytext{its gate symbol}
  \end{minipage}
\end{frame}

\begin{frame}
  An OR gate uses the same parts the other way around. The nMOS go in parallel and the pMOS go in series.\\[3mm]
  \begin{center}
    \graytext{parallel pull-up makes the output \ttab{1} when either input is \ttab{1}}
  \end{center}
  \vspace{2mm}
  \begin{center}
    \graytext{every gate is just a pull-up and a pull-down network of transistors}
  \end{center}
\end{frame}
