    \Large \highlight{\textbf{half adder}}

\begin{frame}
  A \highlight{\textbf{half adder}} adds two bits. It produces a sum bit and a carry bit.\\[3mm]
  \begin{minipage}{0.5\textwidth}
    \centering
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{cc|cc}
      \ttab{a} & \ttab{b} & \ttab{s} & \ttab{c} \\ \hline
      \ttab{0} & \ttab{0} & \ttab{0} & \ttab{0} \\
      \ttab{0} & \ttab{1} & \ttab{1} & \ttab{0} \\
      \ttab{1} & \ttab{0} & \ttab{1} & \ttab{0} \\
      \ttab{1} & \ttab{1} & \ttab{0} & \ttab{1} \\
    \end{tabular}
  \end{minipage}%
  \begin{minipage}{0.5\textwidth}
    \centering
    \graytext{the sum is an XOR} \\[1mm]
    \begin{tikzpicture}
      \node[xor gate US, draw, thick, scale=1.1] (x) at (0,0) {};
      \draw[thick] (x.input 1) -- ++(-0.4,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (x.input 2) -- ++(-0.4,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (x.output) -- ++(0.4,0) node[right, smallgraytext] {\ttab{s}};
    \end{tikzpicture} \\[3mm]
    \graytext{the carry is an AND} \\[1mm]
    \begin{tikzpicture}
      \node[and gate US, draw, thick, scale=1.1] (a) at (0,0) {};
      \draw[thick] (a.input 1) -- ++(-0.4,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (a.input 2) -- ++(-0.4,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (a.output) -- ++(0.4,0) node[right, smallgraytext] {\ttab{c}};
    \end{tikzpicture}
  \end{minipage}
\end{frame}

\begin{frame}{Ripple-Carry Adder}
  \begin{center}
    \Large \highlight{\textbf{ripple-carry adder}}
  \end{center}
\end{frame}

\begin{frame}
  We chain full adders so the carry out of one bit feeds the carry in of the next. This adds two whole numbers.\\[5mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}[scale=0.9]
      \foreach \i/\x in {3/0, 2/2, 1/4, 0/6} {
        \draw[thick, fill=gray!30!white!20] (\x,0) rectangle (\x+1.2,1.1);
        \node[smallgraytext] at (\x+0.6,0.55) {\ttab{FA}};
        \draw[thick] (\x+0.35,1.1) -- ++(0,0.4) node[above, smallgraytext] {\ttab{a\i}};
        \draw[thick] (\x+0.85,1.1) -- ++(0,0.4) node[above, smallgraytext] {\ttab{b\i}};
        \draw[thick] (\x+0.6,0) -- ++(0,-0.4) node[below, smallgraytext] {\ttab{s\i}};
      }
      % carry chain, from right (LSB) to left (MSB)
      \draw[thick, ->] (6,0.55) -- ++(-0.8,0);
      \draw[thick, ->] (4,0.55) -- ++(-0.8,0);
      \draw[thick, ->] (2,0.55) -- ++(-0.8,0);
      \draw[thick, ->] (7.2,0.55) -- ++(0.8,0) node[right, smallgraytext] {\ttab{cin}};
      \draw[thick] (0,0.55) -- ++(-0.8,0) node[left, smallgraytext] {\ttab{cout}};
    \end{tikzpicture}
  \end{figure}
  \begin{center}
    \graytext{the carry ripples from the lowest bit to the highest}
  \end{center}
\end{frame}

% subtraction

\begin{frame}{Subtraction}
  \begin{center}
    \Large \highlight{\textbf{subtraction}}
  \end{center}
\end{frame}

\begin{frame}
  To compute \highlight{a - b}, we add the negation of \ttab{b}. Recall that negating means inverting the bits and adding one.\\[4mm]
  \begin{center}
    $a - b = a + \overline{b} + 1$
  \end{center}
  \vspace{3mm}
  \begin{center}
    \graytext{we invert each bit of \ttab{b} and set the carry in to \ttab{1}}
  \end{center}
\end{frame}

\begin{frame}
  The same adder now does both operations. One control bit selects add or subtract.\\[4mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}
      \draw[thick, fill=gray!30!white!20] (0,-1.2) rectangle (2.4,1.2);
      \node at (1.2,0) {\small adder};
      \draw[thick] (0,0.6) -- ++(-0.8,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (0,-0.6) -- ++(-0.8,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (1.2,-1.2) -- ++(0,-0.5) node[below, smallgraytext] {\ttab{sub}};
      \draw[thick] (2.4,0) -- ++(0.8,0) node[right, smallgraytext] {\ttab{result}};
    \end{tikzpicture}
  \end{figure}
  \begin{center}
    \graytext{\ttab{sub}$=$\ttab{0} adds, \ttab{sub}$=$\ttab{1} subtracts}
  \end{center}
\end{frame}

\begin{frame}
  An \highlight{\textbf{arithmetic logic unit}} gathers these operations into one block. An operation select input chooses what it computes.\\[4mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}
      \draw[thick, fill=gray!30!white!20]
        (0,1.4) -- (2.4,0.6) -- (2.4,-0.6) -- (0,-1.4) -- (0,-0.4) -- (0.5,0) -- (0,0.4) -- cycle;
      \node at (1.2,0) {\small ALU};
      \draw[thick] (0.28,1.05) -- ++(-1.0,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (0.28,-1.05) -- ++(-1.0,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (1.2,1.0) -- ++(0,0.5) node[above, smallgraytext] {\ttab{op}};
      \draw[thick] (2.4,0) -- ++(0.8,0) node[right, smallgraytext] {\ttab{result}};
    \end{tikzpicture}
  \end{figure}
  \begin{center}
    \graytext{the same hardware adds, subtracts, and computes logic operations}
  \end{center}
\end{frame}

\begin{frame}
  \begin{center}
    The ALU is the part of a processor that does the actual computing. Next we add memory so results can be stored.
  \end{center}
\end{frame}
