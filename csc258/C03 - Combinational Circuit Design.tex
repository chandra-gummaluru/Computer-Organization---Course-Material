  \begin{figure}

    \end{tikzpicture}

% ------------------------------------------------------------
% sum of products
% ------------------------------------------------------------
\begin{frame}{Sum of Products}
  \begin{center}
    \Large \highlight{\textbf{sum of products}}
  \end{center}
\end{frame}

\begin{frame}
  A \highlight{\textbf{minterm}} is an AND of every input, each in true or negated form, that is \ttab{1} for exactly one row of the truth table.\\[4mm]
  \begin{center}
    \graytext{for inputs \ttab{a} and \ttab{b}, the row \ttab{a}$=$\ttab{1}, \ttab{b}$=$\ttab{0} has minterm}\\[2mm]
    \highlight{a \cdot \overline{b}}
  \end{center}
\end{frame}

\begin{frame}
  To build a circuit, we OR together the minterm of every row whose output is \ttab{1}.\\[3mm]
  \begin{minipage}{0.5\textwidth}
    \centering
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{cc|c}
      \ttab{a} & \ttab{b} & \ttab{y} \\ \hline
      \ttab{0} & \ttab{0} & \ttab{0} \\
      \ttab{0} & \ttab{1} & \ttab{1} \\
      \ttab{1} & \ttab{0} & \ttab{1} \\
      \ttab{1} & \ttab{1} & \ttab{0} \\
    \end{tabular}
  \end{minipage}%
  \begin{minipage}{0.5\textwidth}
    \centering
    \graytext{two rows give \ttab{1}} \\[3mm]
    $y = \overline{a} \cdot b + a \cdot \overline{b}$
  \end{minipage}
\end{frame}

\begin{frame}
  This expression maps directly to gates. Each product is an AND, and the sum is a single OR.\\[4mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}
      \node[and gate US, draw, thick, scale=1.4] (g1) at (0,1.2) {};
      \node[and gate US, draw, thick, scale=1.4] (g2) at (0,-1.2) {};
      \node[or gate US, draw, thick, scale=1.4] (o) at (3,0) {};

      \draw[thick] (g1.input 1) -- ++(-0.6,0) node[left, smallgraytext] {$\overline{a}$};
      \draw[thick] (g1.input 2) -- ++(-0.6,0) node[left, smallgraytext] {\ttab{b}};
      \draw[thick] (g2.input 1) -- ++(-0.6,0) node[left, smallgraytext] {\ttab{a}};
      \draw[thick] (g2.input 2) -- ++(-0.6,0) node[left, smallgraytext] {$\overline{b}$};

      \draw[thick] (g1.output) -| (o.input 1);
      \draw[thick] (g2.output) -| (o.input 2);
      \draw[thick] (o.output) -- ++(0.6,0) node[right, smallgraytext] {\ttab{y}};
    \end{tikzpicture}
  \end{figure}
\end{frame}

% minimization

\begin{frame}{Minimization}
  \begin{center}
    \Large \highlight{\textbf{minimization}}
  \end{center}
\end{frame}

\begin{frame}
  A sum of products works, but it is rarely the smallest circuit. Fewer terms means fewer gates, less space, and less delay.\\[3mm]
  \begin{center}
    \graytext{we want a systematic way to find a small expression}
  \end{center}
\end{frame}

% ------------------------------------------------------------
% karnaugh maps
% ------------------------------------------------------------
\begin{frame}{Karnaugh Maps}
  \begin{center}
    \Large \highlight{\textbf{Karnaugh maps}}
  \end{center}
\end{frame}

\begin{frame}
  A \highlight{\textbf{Karnaugh map}} rewrites the truth table as a grid. Neighbouring cells differ in exactly one input, so their column labels follow a special order.\\[4mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}[scale=0.9]
      \draw[thick] (0,0) grid (4,2);
      \foreach \x/\l in {0.5/00, 1.5/01, 2.5/11, 3.5/10}
        \node[smallgraytext] at (\x,2.35) {\ttab{\l}};
      \node[smallgraytext] at (-0.45,1.5) {\ttab{0}};
      \node[smallgraytext] at (-0.45,0.5) {\ttab{1}};
      \node[smallgraytext] at (2,2.9) {\ttab{bc}};
      \node[smallgraytext] at (-0.9,1) {\ttab{a}};
    \end{tikzpicture}
  \end{figure}
  \vspace{1mm}
  \begin{center}
    \graytext{the columns step \ttab{00}, \ttab{01}, \ttab{11}, \ttab{10}, changing one bit at a time}
  \end{center}
\end{frame}

\begin{frame}
  We fill each cell with the output for that input. Here is the map for one function of \ttab{a}, \ttab{b}, and \ttab{c}.\\[3mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}[scale=0.9]
      \draw[thick] (0,0) grid (4,2);
      \foreach \x/\l in {0.5/00, 1.5/01, 2.5/11, 3.5/10}
        \node[smallgraytext] at (\x,2.35) {\ttab{\l}};
      \node[smallgraytext] at (-0.45,1.5) {\ttab{0}};
      \node[smallgraytext] at (-0.45,0.5) {\ttab{1}};
      \node[smallgraytext] at (2,2.9) {\ttab{bc}};
      \node[smallgraytext] at (-0.9,1) {\ttab{a}};
      % top row a=0
      \node at (0.5,1.5) {\ttab{1}}; \node at (1.5,1.5) {\ttab{1}};
      \node at (2.5,1.5) {\ttab{0}}; \node at (3.5,1.5) {\ttab{0}};
      % bottom row a=1
      \node at (0.5,0.5) {\ttab{1}}; \node at (1.5,0.5) {\ttab{1}};
      \node at (2.5,0.5) {\ttab{1}}; \node at (3.5,0.5) {\ttab{0}};
    \end{tikzpicture}
  \end{figure}
\end{frame}

\begin{frame}
  We circle the largest groups of adjacent \ttab{1}s whose size is a power of two. Each group becomes one product term.\\[2mm]
  \begin{figure}
    \centering
    \begin{tikzpicture}[scale=0.9]
      \draw[thick] (0,0) grid (4,2);
      \foreach \x/\l in {0.5/00, 1.5/01, 2.5/11, 3.5/10}
        \node[smallgraytext] at (\x,2.35) {\ttab{\l}};
      \node[smallgraytext] at (-0.45,1.5) {\ttab{0}};
      \node[smallgraytext] at (-0.45,0.5) {\ttab{1}};
      \node[smallgraytext] at (2,2.9) {\ttab{bc}};
      \node[smallgraytext] at (-0.9,1) {\ttab{a}};
      \node at (0.5,1.5) {\ttab{1}}; \node at (1.5,1.5) {\ttab{1}};
      \node at (2.5,1.5) {\ttab{0}}; \node at (3.5,1.5) {\ttab{0}};
      \node at (0.5,0.5) {\ttab{1}}; \node at (1.5,0.5) {\ttab{1}};
      \node at (2.5,0.5) {\ttab{1}}; \node at (3.5,0.5) {\ttab{0}};
      % group of four: columns 00,01 both rows -> b bar
      \draw[very thick, myblue, rounded corners=4pt] (0.12,0.12) rectangle (1.88,1.88);
      \node[myblue, smallgraytext] at (1,-0.45) {$\overline{b}$};
      % group of two: bottom row columns 01,11 -> a c
      \draw[very thick, myorange, rounded corners=4pt] (1.12,0.12) rectangle (2.88,0.88);
      \node[myorange, smallgraytext] at (2,-0.45) {$a \cdot c$};
    \end{tikzpicture}
  \end{figure}
  \begin{center}
    $y = \overline{b} + a \cdot c$
  \end{center}
\end{frame}

\begin{frame}
  \begin{center}
    Grouping turns four minterms into two small terms, giving a circuit with far fewer gates.
  \end{center}
\end{frame}
